66 std::string register_subscription(std::string topic_name, std::string subscriber_name, std::function<void(std::shared_ptr<msg_type>)> to_register)
67 {
68 // grab the mutex first
69 std::lock_guard<std::mutex> lock(this->register_mutex);
70
71 // if it's an empty name then assign it a unique identifier
72 std::string local_subscriber_name = subscriber_name;
73
74 if ( not strcmp(local_subscriber_name.c_str(), "") )
75 {
76 local_subscriber_name = topic_name + "_subscriber_" + std::to_string(this->unique_id);
77 this->unique_id++;
78 }
79
80 // we have not yet registered a subscriber to this topic so we initialize the first one
81 if ( this->function_tables.find(topic_name) == this->function_tables.end() )
82 {
83 std::vector<std::map<std::string, std::function<void(std::shared_ptr<msg_type>)>>> n_function_table{{{local_subscriber_name, to_register}}};
84 this->function_tables[topic_name] = n_function_table;// realistically we should statically size this to the max to avoid copy constructor slow downs ( for now it should be fine )
85
86 std::function<void(std::shared_ptr<msg_type>)> n_cb = std::bind(&CommunicationAggregator::generic_subscriber_callback<msg_type>, this, topic_name, std::placeholders::_1, 0);
87 std::vector<std::shared_ptr<rclcpp::Subscription<msg_type>>> n_subscription = {this->create_subscription<msg_type>(topic_name, this->sub_qos, n_cb)};
88
89 this->subscription_ptrs[topic_name] = n_subscription;
90
91 return local_subscriber_name;
92 }
93
95 // std::map<std::string, std::function<void(std::shared_ptr<msg_type>)>>
96 auto tmp_func_vector = std::any_cast< std::vector< std::map< std::string, std::function< void( std::shared_ptr<msg_type> ) > > > > ( this->function_tables[topic_name] );
97
98 // find the first available index
99 int n_index = 0;
100
101 for ( ; n_index < (int) tmp_func_vector.size() ; n_index++ )
102 if ( (int) tmp_func_vector[n_index].size() < SUBSCRIBER_CB_MAX )
103 break;
104
105 // found an existing subscriber with room
106 if ( n_index < (int) tmp_func_vector.size() )
107 {
108 tmp_func_vector[n_index][local_subscriber_name] = to_register;
109 }
110 else // have to create a new one
111 {
112 std::function<void(std::shared_ptr<msg_type>)> n_cb = std::bind(&CommunicationAggregator::generic_subscriber_callback<msg_type>, this, topic_name, std::placeholders::_1, n_index);
113 std::shared_ptr<rclcpp::Subscription<msg_type>> n_subscription = this->create_subscription<msg_type>(topic_name, this->sub_qos, n_cb);
114
115 // register the new subscription
116 auto tmp_sub_vector = std::any_cast< std::vector< std::shared_ptr< rclcpp::Subscription< msg_type > > > > (this->subscription_ptrs[topic_name]);
117 tmp_sub_vector.push_back(n_subscription);
118
119 this->subscription_ptrs[topic_name] = tmp_sub_vector;
120
121 tmp_func_vector.push_back({{local_subscriber_name, to_register}});
122 }
123
124 this->function_tables[topic_name] = tmp_func_vector;
125
126 return local_subscriber_name;

142 template <typename msg_type>
143 bool unregister_subscription(std::string topic_name, std::string subscriber_name = "")
144 {
145 // grab the mutex first
146 std::lock_guard<std::mutex> lock(this->register_mutex);
147
148 if ( this->function_tables.find(topic_name) == this->function_tables.end() )
149 return false;
150
151 auto tmp_func_vector = std::any_cast< std::vector < std::map < std::string, std::function< void( std::shared_ptr< msg_type > ) > > > > (this->function_tables[topic_name]);
152 auto tmp_sub_vector = std::any_cast< std::vector< std::shared_ptr< rclcpp::Subscription< msg_type > > > > (this->subscription_ptrs[topic_name]);
153
154 if ( not strcmp(subscriber_name.c_str(), "") )
155 {
156 // deleting everything is easy doing the any cast first to ensure that destructors are called
157 tmp_func_vector.clear();
158 this->function_tables.erase(topic_name);
159
160 tmp_sub_vector.clear();
161 this->subscription_ptrs.erase(topic_name);
162
163 return true;
164 }
165
166 // look in each of the different dictionaries for the guy
167 typename std::vector<std::map<std::string, std::function< void( std::shared_ptr<msg_type> ) > > >::iterator to_delete = tmp_func_vector.begin();
168
169 for ( ; to_delete != tmp_func_vector.end() ; to_delete++ )
170 {
171 if ( (*to_delete).find(subscriber_name) != (*to_delete).end() )
172 {
173 (*to_delete).erase(subscriber_name);
174 break;
175 }
176 }
177
178 // if we didn't find the subscriber_name just return
179 if ( to_delete == tmp_func_vector.end() )
180 return false;
181
182 // if the resulting function table is empty free the memory from the subscriber
183 if ( (*to_delete).empty() )
184 {
185 int index = (int) std::distance(tmp_func_vector.begin(), to_delete);
186
187 tmp_func_vector.erase(to_delete);
188 tmp_sub_vector.erase(tmp_sub_vector.begin() + index);
189 }
190
191 // finally save the changes
192 this->function_tables[topic_name] = tmp_func_vector;
193 this->subscription_ptrs[topic_name] = tmp_sub_vector;
194
195 return true;
196 }

464 template <typename msg_type>
465 void generic_subscriber_callback(std::string topic_name, std::shared_ptr<msg_type> msg, int function_table_index)
466 {
467 // in case we unregister the subscriber during this callback being called (Not perfect but this should be safe enough)
468 if ( this->function_tables.find(topic_name) == this->function_tables.end() )
469 return;
470
471 // now cast the function table to the correct type
472 auto table = std::any_cast< std::vector < std::map < std::string , std::function < void(std::shared_ptr< msg_type > ) > > > > (this->function_tables[topic_name])[function_table_index];
473
474 // initialize and start a thread for every registered callback of this subscriber
475 std::list<std::shared_ptr<std::thread>> thread_list;
476 for ( const auto& pair : table )
477 {
478 auto thread_ptr = std::make_shared<std::thread>(pair.second, msg);
479
480 thread_list.push_back(thread_ptr);
481 }
482
483 // join all threads before returning
484 for ( auto iter : thread_list )
485 iter->join();
486 }
487
489 // subscriber members
490 std::map<std::string, std::any> function_tables; // std::any = std::vector < std::map < std::string, std::function< void( std::shared_ptr< msg_type > ) > > >
491 std::map<std::string, std::any> subscription_ptrs; // std::any = std::vector < std::shared_ptr < rclcpp::Subscription < msg_type > > >

493 rclcpp::QoS sub_qos;
